#Sorting 
1.selection sort
IN descending order
#include<bits/stdc++.h>
using namespace std;

void selc(int a[], int n){
    int max;
    for(int i=0; i<n-1; i++){
         max=i;
        for(int j=i+1; j<n; j++){
            if(a[max]<a[j]){
                max=j;
            }
        }
        int temp;
        temp=a[max];
        a[max]=a[i];
        a[i]=temp;
    }
}

int main(){
    int a[]={4,8,9,2,3};
    int n=sizeof(a)/sizeof(a[0]);
    selc(a, n);
    for(int i=0; i<n; i++){
        cout<<a[i]<<" ";
    }
}

2.Bubble sort
#include<bits/stdc++.h>
using namespace std;

void bubb(int a[], int n){
    for(int i=n-1; i>1; i--){
        for(int j=0; j<i; j++){
            if(a[j]<a[j+1]){
                int temp;
                temp=a[j];
                a[j]=a[j+1];
                a[j+1]=temp;
            }
        }
    }
}

int main(){
    int a[]={5,8,9,2,3,7,1,4};
    int n=sizeof(a)/sizeof(a[0]);
    bubb(a, n);
    for(int i=0; i<n; i++){
        cout<<a[i]<<" ";
    }
}

3.Insertion sort
#include<bits/stdc++.h>
using namespace std;

void ins(int a[], int n){
  for(int i=1; i<n; i++){
      int c=a[i];
    while(c>a[i-1] && i!=0){
        a[i]=a[i-1];
        i--;
    }  
    a[i]=c;
  }
}

int main(){
    int a[]={5,8,9,2,4,1,6};
    int n=sizeof(a)/sizeof(a[0]);
    ins(a, n);
    for(int i=0; i<n; i++){
        cout<<a[i]<<" ";
    }
}
